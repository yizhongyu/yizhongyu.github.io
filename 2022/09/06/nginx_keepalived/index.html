Nginx故障转移

Nginx Keepalived
work
发布日期:   2022-09-06
文章字数:   956

使用俩台服务器，服务器A/B各自安装nginx，keepalived

keepalived生成虚拟IP，客户机连接到虚拟IP，虚拟IP可以连接到A/B服务器中的一台

当其中一台的nginx出现问题时，可以跳转到另一台服务器

graph TB
    id1>虚拟IP]--keepalived---id2(服务器A/Nginx) & id3(服务器B/Nginx)
    subgraph  
        id2 & id3
    end
id2 & id3 --代理--- id4[服务1] & id5[服务2]

一：nginx集群

安装nginx与keepalived

yum install -y keepalived nginx
systemctl enable keepalived
systemctl enable nginx

二：keepalived配置

改配置文件(将默认的eth0网卡改为自己的网卡名,绑定当前网卡)

sed -i "s/eth0/ens192/g" /etc/keepalived/keepalived.conf

设置主备(改其中一个为BACKUP并调低优先级)

sed -i "s/MASTER/BACKUP/g" /etc/keepalived/keepalived.conf
sed -i "s/priority 100/priority 90/g" /etc/keepalived/keepalived.conf

配置解释

vrrp_script check_nginx {					#定义一个策略,策略名为"check_nginx"
	script "/etc/keepalived/chech_nginx.sh"	#执行一次/etc/keepalived/check_nginx.sh脚本文件
	interval 5								#每隔5秒钟
	weight -5								#如果发现nginx启动失败了,将当前的机器的priority-5
}

vrrp_instance VI_1 {
    state MASTER				#主MASTER/备BACKUP
    interface ens192			#绑定当前服务器的网卡名
    virtual_router_id 51		#虚拟路由id号，所有机器保持一致
    priority 100				#优先级，主>备
    advert_int 1				#主备之间的通信时间间隔，单位s
    authentication {			#主备之间的通信密码
        auth_type PASS
        auth_pass 1111
    }
    track_script {				#指定keepalived跟踪策略为 "check_nginx"
    	check_nginx
    }
    virtual_ipaddress {			#虚拟IP
        192.168.200.111
    }
}

keepalived与nginx联动-监控

keepalived与nginx关联

vim /etc/keepalived/chech_nginx.sh

#!/bin/bash
	A=`ps -C nginx --no-header |wc -l`        
	if [ $A -eq 0 ];then                            
	    /usr/local/nginx/sbin/nginx                
	    if [ `ps -C nginx --no-header |wc -l` -eq 0 ];then
	        exit 1
	    else
	        exit 0
	    fi
	else
	    exit 0
	fi

检查nginx是否已经启动(ps任务管理器中是否有nginx进程)
IF nginx已经启动，keepalived认为当前机器没有问题，exit 0
IF nginx未启动，keepalived尝试去启动nginx
IF 能够启动成功nginx，keepalived认为当前机器没有问题，exit 0
IF 不能启动成功nginx，keepalived认为当前机器有问题，exit 1
此时nginx不需要我们自己启动，交给keepalived帮助我们去启动它

chmod 744 /etc/keepalived/chech_nginx.sh

定时任务

nginx检查脚本vim /etc/keepalived/chech_nginx.sh

#!/bin/bash
A=`ps -C nginx --no-header |wc -l`
B=`ps -C keepalived --no-header |wc -l`
if [ $A -eq 0 ];then
	/usr/local/nginx/sbin/nginx
	sleep 3
	if [ `ps -C nginx --no-header |wc -l` -eq 0 ];then
		ps -ef|grep keepalived|grep -v|awk '{print $2}'|xargs kill -9
	fi
fi
if [ $A -ne 0 -a $B -eq 0 ];then
	systemctl restart keepalived
fi

添加定时任务

chmod 744 /etc/keepalived/chech_nginx.sh
crontab -e
*/1 * * * * . /etc/keepalived/chech_nginx.sh

nginx反向代理服务停止，keepalived服务也停止

vim /etc/keepalived/check_web.sh

  1. 直接停止

    #!/bin/bash
web_info=$(ps -ef|grep [n]ginx|wc -l)
if [ $web_info -lt 2 ]
then
   /etc/init.d/keepalived stop
fi

vim /etc/keepalived/keepalived.conf
vrrp_script check_web {
    script "/server/scripts/check_web.sh"
    interval 2 
    weight 2 
}
track_script {
     check_web
}

  2. 拯救一次再停止

    #!/bin/bash
web_info_1=$(ps -ef|grep [n]ginx|wc -l)
if [ $web_info_1 -lt 2 ];then
    /etc/init.d/nginx start
    sleep 2
    web_info_2=$(ps -ef|grep [n]ginx|wc -l)
    if [ $web_info_2 -lt 2 ];then
       /etc/init.d/keepalived stop
    fi
fi

vim /etc/keepalived/keepalived.conf
vrrp_script check_web {
    script "/server/scripts/check_web.sh"
    interval 2 
    weight 2 
}
track_script {
     check_web
}

  3. 权重降级

    #!/bin/bash
web_info=$(ps -ef|grep [n]ginx|wc -l)
if [ $web_info -lt 2 ];then
/etc/init.d/nginx start
    return 1
else
    return 0
fi

vim /etc/keepalived/keepalived.conf
vrrp_script check_web {
    script "/server/scripts/check_web.sh"
    interval 2 
    weight 2
}
track_script {
     check_web
}

配置完成后打开keepalived

systemctl start keepalived
