MariaDB基本使用

centos7 MariaDB
数据库
发布日期:   2022-06-07
文章字数:   757

MariaDB是MySQL关系数据库管理系统的一个分支。类似MySQL数据库

查看是否安装
systemctl status mariadb
rpm -qa|grep mariadb
卸载
yum -y remove mariadbxxx
安装
yum install mariadb mariadb-server
启动
systemctl start mariadb
初始化
mysql_secure_installation
密码为空，是否创建新密码	y
密码
移除匿名用户				y
不允许root用户远程登录	y
移除测试数据库				y
重新加载数据库权限		y

登录

mysql -uroot -pmariadbpasswd

创建数据库

show databases;
create database abc;

创建用户123，密码456，每个IP都可访问

create user '123'@'%' identified by '456';

设置数据库专用账号密码赋予用户数据库权限

grant all on abc.* to '123'@'%';

切换数据库

use abc;

创建数据表

show tables;
create table admin
(
username varchar(15) not null unique,
pasword varchar(20) ,
question varchar(20) ,
answer varchar(50)
);
展示列表项
show columns from admin;

增删改查

添加数据
insert into admin values ("admin","123456","","");
查看数据
select * from admin;
更新数据
update admin set question="dadada" where username="admin";
删除数据表中的数据
delete from admin where username="admin";
删除数据表
drop table admin;

打开防火墙指定端口

firewall-cmd --add-port=3306/tcp --zone=public --permanent
systemctl restart firewalld

开机自启

systemctl enable mariadb

脚本安装

#!/bin/bash
# ------------------------------------------
# Filename: mariadb_384.sh
# Revision: 1.1
# Description: yum install mariadb on centos7.4
# Date: 2022-08-01
# Author: clarence
gridknow=/usr/local/gridknow
username=root
password=123456
#安装服务并拷贝配置文件
    yum -y install mariadb-server mariadb
echo "[client]
port            = 3306
default-character-set = utf8

[mysqld]
datadir=/var/lib/mysql
socket=/var/lib/mysql/mysql.sock

#add
port            = 3306
skip-name-resolve
key_buffer_size = 96M
table_cache = 256
sort_buffer_size = 3M
join_buffer_size = 3M
net_buffer_length = 8K
read_buffer_size = 3M
read_rnd_buffer_size = 1M
max_allowed_packet = 128M
max_heap_table_size = 32M
tmp_table_size = 512M
back_log = 100
default-storage-engine=MyISAM
wait_timeout = 3880000
interactive_timeout = 3880000

thread_cache_size = 256
thread_concurrency = 8

log_bin_trust_function_creators = 1
event_scheduler = 1
binlog_cache_size = 1M
lower_case_table_names = 1
character_set_server = utf8
max_connections = 4000
max_connect_errors = 1000000
max_prepared_stmt_count = 1048575
wait_timeout = 3880000
interactive_timeout = 3880000
query_cache_size = 128M
query_cache_limit = 2M
expire_logs_days = 7

log-bin = mysql-bin
log-bin-index = binlog.index
server-id = 33061
binlog_format = ROW
relay-log = mysql-relay-bin

innodb_buffer_pool_size = 1024M
innodb_additional_mem_pool_size = 2M
innodb_log_file_size = 64M
innodb_log_buffer_size = 8M

[mysqldump]
quick

[mysql]
no-auto-rehash

[isamchk]

read_buffer = 2M

[myisamchk]
key_buffer = 20M
sort_buffer_size = 20M
read_buffer = 2M
write_buffer = 2M

[mysqlhotcopy]
interactive-timeout = 7200
# Disabling symbolic-links is recommended to prevent assorted security risks
symbolic-links=0
# Settings user and group are ignored when systemd is used.
# If you need to run mysqld under a different user or group,
# customize your systemd unit file for mariadb according to the
# instructions in http://fedoraproject.org/wiki/Systemd

[mysqld_safe]
log-error=/var/log/mariadb/mariadb.log
pid-file=/var/run/mariadb/mariadb.pid

#
# include all files from the config directory
#
!includedir /etc/my.cnf.d

" >/etc/my.cnf
#启动服务并设置为开机自动启动
    systemctl start mariadb
    systemctl enable mariadb
#配置密码；创建数据库；导入数据库；修改权限
    mysqladmin -u $username password $password
    mysql -u${username} -p${password} -e "
	create database cdms;
	create database datamanage2;
	create database userinternet_record;
	create database alertmsg;
	create database zkconfig;
	create database apgs3;
	create database iteamwebnew;
	create database appstore;
	create database pmchat;
	create database ephemeris;
	create database pmdashboard;
	create database gk_user_center;
	create database pmoa;
	create database gkrm;
	create database produce-data;
	create database gkskdsvr;
	create database push_server;
	create database gktomcat;
	GRANT ALL PRIVILEGES ON *.* TO 'root'@'%' identified by 'mariadbpasswd' with grant option;
	flush privileges;
	quit "
sed -i "34i LimitNOFILE=10000" /usr/lib/systemd/system/mariadb.service
sed -i "35i LimitNPROC=10000" /usr/lib/systemd/system/mariadb.service
systemctl --system daemon-reload
systemctl restart mariadb

echo '------------mariadb has been installed and completed------------'
